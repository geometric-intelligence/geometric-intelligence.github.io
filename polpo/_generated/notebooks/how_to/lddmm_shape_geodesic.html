def compute_max_vertex_abs_diff(mesh_dict, cmp_mesh_dict):
    keys = set(mesh_dict.keys()).intersection(set(cmp_mesh_dict.keys()))

    diffs = [
        np.amax(np.abs(mesh_dict[key].points - cmp_mesh_dict[key].points))
        for key in keys
    ]

    return np.amax(diffs)
