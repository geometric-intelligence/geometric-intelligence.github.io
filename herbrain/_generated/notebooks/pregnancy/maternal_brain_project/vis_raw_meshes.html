3.1.1. Load enigma meshes#

subject_subset = ["01"] if DEBUG else None
session_subset = list(range(1, 6)) if DEBUG else None
derivative = "enigma"

mesh_loader = MeshLoader(
    subject_subset=subject_subset,
    struct_subset=structs,
    session_subset=session_subset,
    as_mesh=True,
    derivative=derivative,
) + ppdict.DictMap(ppdict.NestedDictSwapper())

# subject, struct, session
meshes = mesh_loader()

[6]:

for subject_id, meshes_ in meshes.items():
    pl = RegisteredMeshesGifPlotter(
        shape=putils.plot_shape_from_n_plots(len(meshes_), n_axis=2, axis=0),
        rowise=False,
        fps=5,
        border=False,
        off_screen=True,
        notebook=False,
        window_size=(1920, 1080),
        subtitle=(
            lambda session_id, struct_name: f"{subject_id}, {struct_name}, {session_id}"
        ),
    )

    pl.add_meshes(ppdict.NestedDictSwapper()(meshes_))
    pl.close()

    pl.show()

    RigidAlignment(known_correspondences=True),
    depth=1,
)

meshes = prep_pipe(raw_meshes)
